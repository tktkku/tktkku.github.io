0day安全-栈溢出初探

0x01 实验环境

  • 操作系统： Windows xp sp2
  • 编译器：Visual C++ 6.0
  • 编译选项：默认
  • build版本：debug

0x02 实验源程序

#include <stdio.h>
#include <windows.h>
#define PASSWORD "1234567"
int verify_password(char* password)
{
    int authenticated;
    char buffer[44];
    authenticated = strcmp(password, PASSWORD);
    strcpy(buffer, password);
    return authenticated;
}
int main()
{
	int valid_flag = 0;
    char password[1024];
    FILE* fp;
    LoadLibrary("user32.dll");
    if (!(fp = fopen("password.txt", "rw+")))
    {
        exit(0);
    }
    fscanf(fp, "%s", password);
    valid_flag = verify_password(password);
    if (valid_flag)
    {
        printf("incorrect password!\n");
    }
    else
    {
        printf("Congratulation! You have passed the verification！\n");
    }
    fclose(fp);
    return 0;
}

0x03 实验原理

strcpy函数不检查字符串长度，当长度超过分配的空间就会覆盖掉其他栈帧，向下覆盖掉EIP位特定的指令地址，达到执行任意代码的效果。

假设有这样一段汇编程序：

push b
push a
call func

;func
push ebp
mov ebp,esp
sub esp,8
mov [ebp-8],0
;其他操作
add esp,8
ret 8

它所对应的栈：

如果可供使用的空间足够大，直接把局部变量覆盖为shellcode，返回地址处覆盖为局部变量（shellcode）的起始位置，但是当堆栈环境不一样时shellcode的地址也会变化，适用性很小：

还有一种就是把shellcode放到后面去，esp就是shellcode的起始位置了，把返回地址覆盖为jmp esp，这样不管怎么变，都能定位到，但是也有可能有些地址是不可写的：

0x04 实验步骤

一、找jmp esp地址

利用程序：

#include <windows.h>
#include <stdio.h>
#define DLL_NAME "user32.dll"
int main()
{
    BYTE* ptr;
    int position, address;
    HINSTANCE handle;
    BOOL done_flag = FALSE;
    handle = LoadLibrary(DLL_NAME);
    if (!handle)
    {
        printf(" load dll erro !");
        exit(0);
    }
    ptr = (BYTE*)handle;
    for (position = 0; !done_flag; position++)
    {
		try
        {
            // 0xFFE4是jmp esp的opcode
            if (ptr[position] == 0xff && ptr[position+1] == 0xe4)
            {
                int address = (int)ptr + position;
                printf("OPCODE found at 0x%x\n", address);
            }
        }
        catch(...)
        {
            int address = (int)ptr + position;
            printf("END OF 0x%x\n", address);
            done_flag = true;
        }
    }
}
