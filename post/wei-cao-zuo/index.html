b&1运算可以求得b的最低位，b>>1可以舍去最低位。

#include<stdio.h>
int power(int a, int b, int p) {
    int res = 1 % p;
    for (; b; b >>=1) {
        if (b & 1) {
            res = (long long) res * a % p;
        }
        a = (long long) a * a % p;
    }
    return res;
}
int main() {
    int a, b, p;
    while (scanf("%d %d %d", &a, &b, &p) != EOF) {
        printf("%d\n", power(a, b, p));
    }
}

#include<stdio.h>
long long mul(long long a, long long b, long long p) {
    long long res = 0;
    for (; b; b >>=1) {
        if (b & 1) {
            res = (res + a) % p;
        }
        a = a * 2 % p;
    }
    return res;
}
